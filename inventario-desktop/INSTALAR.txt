                    INVENTARIO PROFESIONAL v1.0
                         Guía de Instalación
═══════════════════════════════════════════════════════════════

🎯 ¿QUÉ ES ESTO?
Una aplicación de escritorio para gestionar tu inventario.
No necesita navegador, funciona como cualquier programa.

📦 OPCIONES DE INSTALACIÓN

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
OPCIÓN 1: EJECUTAR DIRECTAMENTE (Recomendado)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Busca el archivo: Inventario-Profesional-Setup-1.0.0.exe
2. Haz doble clic para ejecutar
3. Sigue las instrucciones del instalador
4. Listo! Busca "Inventario Profesional" en tu menú de inicio

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
OPCIÓN 2: SI NO TIENES EL EJECUTABLE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Asegúrate de tener Node.js instalado
   • Si no lo tienes, descárgalo desde: https://nodejs.org/
   • Descarga la versión "LTS" (la recomendada)

2. Ejecuta el archivo start.bat
   • Haz doble clic en start.bat
   • Espera a que termine el proceso
   • Esto creará el instalador automáticamente

3. Sigue los pasos de la OPCIÓN 1

🚀 CÓMO USAR LA APLICACIÓN

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
AGREGAR PRODUCTOS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Click en "Agregar Producto"
• Nombre: Ej: "Laptop Dell"
• Categoría: Ej: "Electrónicos"
• Cantidad: Número inicial
• Precio: OPCIONAL (puedes dejarlo vacío)
• Notas: OPCIONAL

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
AJUSTAR CANTIDAD (FÁCIL)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Botón VERDE (+) = Aumenta en 1
• Botón ROJO (-) = Disminuye en 1
• No necesitas editar, solo click en los botones

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
ALERTA "NECESITAMOS"
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Cuando un producto llega a 0, aparece en rojo: "NECESITAMOS"
• Estos productos se muestran primero en la lista
• Te avisa automáticamente cuando algo se agota

💾 RESPALDO DE TUS DATOS

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
EXPORTAR (COPIA DE SEGURIDAD)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Click en "Exportar"
• Guarda el archivo en un lugar seguro
• Puedes importarlo después si lo necesitas

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
¿DÓNDE SE GUARDAN LOS DATOS?
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Los datos se guardan en la carpeta de la aplicación
• Archivo: data.json
• Para hacer respaldo: solo copia este archivo

🆘 PROBLEMAS COMUNES

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
❌ "NO SE PUEDE ABRIR EL PROGRAMA"
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Asegúrate de tener Windows 10 o superior
• Haz click derecho → "Ejecutar como administrador"
• Revisa que tu antivirus no lo esté bloqueando

❌ "SE CIERRA SOLAMENTE"
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Reinstala el programa
• Asegúrate de tener permisos en la carpeta

❌ "NO GUARDA LOS DATOS"
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
• Ejecuta como administrador
• Revisa que puedas escribir en la carpeta de instalación
